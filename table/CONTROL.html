TRIGGER calcular_riesgo_inherente
BEFORE INSERT OR UPDATE ON control
FOR EACH ROW
BEGIN
    :NEW.riesgo_inherente := TO_CHAR(:NEW.controlescala_frecuencia * :NEW.controlescala_impacto);
END;
TRIGGER "LIMITAR_VALORES" BEFORE INSERT OR UPDATE ON control
FOR EACH ROW
BEGIN
  IF :NEW.periocidad > 5 THEN
    :NEW.periocidad := 5;
  END IF;
  IF :NEW.responsabilidad > 5 THEN
    :NEW.responsabilidad := 5;
  END IF;
  IF :NEW.eficacia > 5 THEN
    :NEW.eficacia := 5;
  END IF;
  IF :NEW.documental > 5 THEN
    :NEW.documental := 5;
  END IF;
  IF :NEW.oportunidad > 5 THEN
    :NEW.oportunidad := 5;
  END IF;
  IF :NEW.automatizacion > 5 THEN
    :NEW.automatizacion := 5;
  END IF;
END;
TRIGGER "CALCULAR_CONTROL_IMPACTO" BEFORE
    INSERT OR UPDATE OF documental, oportunidad, automatizacion ON control
    FOR EACH ROW
BEGIN
    :new.controlescala_impacto := to_char(:new.documental * :new.oportunidad * :new.automatizacion);
END;
TRIGGER "CALCULAR_CONTROL_FRECUENCIA" BEFORE INSERT OR UPDATE OF periocidad, eficacia, responsabilidad ON control
FOR EACH ROW
BEGIN
    :NEW.controlescala_frecuencia := TO_CHAR(:NEW.periocidad * :NEW.responsabilidad * :NEW.eficacia);
END;
TRIGGER "ASIGNAR_IMPACTO" AFTER
    INSERT OR UPDATE ON CONTROL
    FOR EACH ROW
BEGIN
    UPDATE "USER_PROYECTO"."Mapa_Calor"
    SET
        "CONTROL_IMPACTO" =
            CASE
            --Crea una nueva columna con los valores:
                WHEN :new.controlescala_impacto = 'Catastrofico'
                     AND :new.eficacia = 'Alta' THEN
                    'Alta'
                WHEN :new.controlescala_impacto = 'Mayor'
                     AND :new.eficacia = 'Alta' THEN
                    'Medio'
                WHEN :new.controlescala_impacto = 'Moderado'
                     AND :new.eficacia = 'Alta' THEN
                    'Baja'
                WHEN :new.controlescala_impacto = 'Menor'
                     AND :new.eficacia = 'Media' THEN
                    'Media'
                WHEN :new.controlescala_impacto = 'Insignificante'
                     AND :new.eficacia = 'Baja' THEN
                    'Baja'
                ELSE
                    'No determinado'
            END
    WHERE
        "ID_MAPACALOR" = :new.MAPA_DE_CALOR_ID_MAPACALOR;

END;
